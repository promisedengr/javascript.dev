function manipulateStudentRecord(obj, operation, prop, newValue) {
    if(operation === 'edit') {
        if(obj[prop] !== undefined) {
            obj[prop] = newValue;
        }
        return obj;
    }
    if(operation === 'delete') {
        delete obj[prop];
        return obj;
    }
    return obj;
}
